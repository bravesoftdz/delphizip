OnProgress event

Applies to

TZipMaster component

Declaration

property OnProgress: TZMProgressEvent;

type
  TZMProgressEvent = procedure(Sender: TObject; details: TZMProgressDetails) of object;

Arguments
Sender: TObject
The component firing the event
details: TZMProgressDetails
A read-only class of the details
Description

OnProgress event occurs during operations on the Zip archive.

(See Demo1 to learn how to use this event.)

Example
procedure TForm1.ZipProgress(Sender: TObject; details: TZMProgressDetails);
var
  s: String;
begin
  case details.Order of
    NewFile:
      begin
        Label1.Caption := details.ItemName;
        ProgressBar1.Position := 0;   // item percent
        ProgressBar3.Position := details.ItemNumber;  // files count
        s := 'File = ' + IntToStr(details.ItemNumber) + ' of ' + IntToStr(details.TotalCount);
        Label19.Caption := s;
      end;
    NewExtra:
      begin
        Label1.Caption := details.ItemName;
        ProgressBar1.Position := 0;
      end;
    EndOfItem:;
    EndOfBatch:
      begin
        Label1.Caption := '';
        ProgressBar1.Position := 0;
        Label2.Caption := '';
        ProgressBar2.Position := 0;  // total percent
        ProgressBar3.Position := 0;
        ProgressBar3.Max := 100;
      end;
    TotalFiles2Process:
      begin
        ProgressBar3.Max := details.TotalCount;
      end;
    TotalSize2Process:
      begin
        Label2.Caption := IntToStr(details.TotalSize);
        ProgressBar2.Position := 0;
      end;
    ProgressUpdate:
      begin
        ProgressBar1.Position := details.ItemPerCent;
        ProgressBar2.Position := details.TotalPerCent;
        s := 'Total = ' + IntToStr(details.TotalSize)
          + ' Position = ' + IntToStr(details.TotalPosition)
          + ' Written = ' + IntToStr(details.BytesWritten);
        Label2.Caption := s;
      end;
    ExtraUpdate:
      begin
        ProgressBar1.Position := details.ItemPerCent;
        ProgressBar2.Position := details.TotalPerCent;
      end;
  end;
end;
