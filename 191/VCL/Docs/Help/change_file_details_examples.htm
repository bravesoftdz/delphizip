 // if 'data' = 1 renames with '6' replaced by '_6'
 function CFDfunc1(theRec: TZMDirRec; var Data): Integer;
 var
   iData: integer Absolute Data;
 begin
   if iData = 1 then
     Result := theRec.ChangeName(StringReplace(theRec.FileName, '6', '_6', [rfReplaceAll]))
   else
     Result := 1; // nothing done
 end;

 function TForm1.ChangeName: Integer;
 var
   data: Integer;
 begin
   data := 1;
   Zip1.FSpecArgs.Add('*6.txt');
   Result := Zip1.ChangeFileDetails(CFDfunc1, data);
 end;

 // Change the file date
 function CFDfunc2(theRec: TZMDirRec; var Data): Integer;
 var
   iData: Cardinal Absolute Data;
 begin
   Result := theRec.ChangeDate(iData);
 end;

 function TForm1.ChangeDate: Integer;
 var
   data: Cardinal;
   testDate: TDateTime;
 begin
   testDate := StrToDateTime('10/04/2003 3:14');
   data := DateTimeToFileDate(testDate);
   Zip1.FSpecArgsExcl.Clear;
   Zip1.FSpecArgs.Clear;
   Zip1.FSpecArgs.Add('*d*.txt');
   Result := Zip1.ChangeFileDetails(CFDfunc2, data);
 end;

 // Change the file comment
 function CFDfunc3(theRec: TZMDirRec; var Data): Integer;
 var
   iData: String Absolute Data;
 begin
   Result := theRec.ChangeComment(iData);
 end;

 function TForm1.ChangeComment: Integer;
 var
   Comment: String;
 begin
   Comment := 'some comment';
   Zip1.FSpecArgsExcl.Clear;
   Zip1.FSpecArgs.Clear;
   Zip1.FSpecArgs.Add('*a*.txt');
   Result := Zip1.ChangeFileDetails(CFDfunc3, Comment);
 end;

 // Change the file attribute
 function CFDfunc4(theRec: TZMDirRec; var Data): Integer;
 var
   iData: DWORD Absolute Data;
 begin
   Result := theRec.ChangeAttrs(iData);
 end;


 function TForm1.ChangeAttributes: Integer;
 var
   newAttr: DWORD;
 begin
   newAttr := FILE_ATTRIBUTE_READONLY;
   Zip1.FSpecArgsExcl.Clear;
   Zip1.FSpecArgs.Clear;
   Zip1.FSpecArgs.Add('*l*.txt');
   Result := Zip1.ChangeFileDetails(CFDfunc4, newAttr);
 end;

 procedure TForm1.Button1Click(Sender: TObject);
 var
   err: Integer;
 begin
    Zip1.Clear;
    Zip1.ZipFileName := 'zTest.zip';
    if Zip1.Count > 1 then
    begin
      err := ChangeName;
      Memo1.Lines.Add(Format(' ChangeName returns %d [ %d files ]', [err, Zip1.SuccessCnt]));
      if err <> 0 then
        Memo1.Lines.Add(' failed ' + Zip1.ErrMessage)
      else
      begin
        Memo1.Lines.Add('Changed files ----');
        Memo1.Lines.AddStrings(Zip1.FSpecArgs);
        Memo1.Lines.Add('Skipped files ----');
        Memo1.Lines.AddStrings(Zip1.FSpecArgsExcl);
      end;        

      err := ChangeDate;
      Memo1.Lines.Add(Format(' ChangeDate returns %d [ %d files ]', [err, Zip1.SuccessCnt]));
      if err <> 0 then
        Memo1.Lines.Add(' failed ' + Zip1.ErrMessage);
      err := ChangeComment;
      Memo1.Lines.Add(Format(' ChangeComment returns %d [ %d files ]', [err, Zip1.SuccessCnt]));
      if err <> 0 then
        Memo1.Lines.Add(' failed ' + Zip1.ErrMessage);
      err := ChangeAttributes;
      Memo1.Lines.Add(Format(' ChangeAttributes returns %d [ %d files ]', [err, Zip1.SuccessCnt]));
      if err <> 0 then       
         Memo1.Lines.Add(' failed ' + Zip1.ErrMessage);
    end;
 end;
